Introduction to JavaScript

JavaScript is a versatile, high-level programming language that makes web pages interactive. It's one of the three core technologies of web development, alongside HTML and CSS.

What is JavaScript?

JavaScript provides:

  • Interactivity: Respond to user actions
  • Dynamic Content: Update content without page reloads
  • Browser APIs: Access browser features and hardware
  • Server-Side: Node.js enables backend development
  • Frameworks: React, Angular, Vue for complex applications
ES6+ (ECMAScript 2015+): Modern JavaScript includes arrow functions, classes, modules, promises, and many other improvements.

// ES6 Classes
class Person {
  // Constructor
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  
  // Instance method
  greet() {
    return `Hello, my name is ${this.name}`;
  }
  
  // Static method
  static speciesInfo() {
    return "Humans are Homo sapiens";
  }
  
  // Getter
  get description() {
    return `${this.name} is ${this.age} years old`;
  }
  
  // Setter
  set newAge(age) {
    if (age > 0) {
      this.age = age;
    } else {
      console.log("Age must be positive");
    }
  }
}

// Creating instances
const person1 = new Person("Alice", 25);
const person2 = new Person("Bob", 30);

console.log(person1.greet()); // Hello, my name is Alice
console.log(Person.speciesInfo()); // Static method call
console.log(person1.description); // Getter
person1.newAge = 26; // Setter

// Inheritance
class Student extends Person {
  constructor(name, age, major) {
    super(name, age); // Call parent constructor
    this.major = major;
  }
  
  study() {
    return `${this.name} is studying ${this.major}`;
  }
  
  // Override parent method
  greet() {
    return super.greet() + " and I'm a student";
  }
}

const student = new Student("Charlie", 20, "Computer Science");
console.log(student.greet());
console.log(student.study());

  setTimeout(resolve, 100, "foo");
});

  .then(values => {
    console.log(values); // [3, 42, "foo"]
  });

// Promise.race() - First promise to settle wins
const fastPromise = new Promise(resolve => 
  setTimeout(resolve, 500, "Fast")
);
const slowPromise = new Promise(resolve => 
  setTimeout(resolve, 1000, "Slow")
);

Promise.race([fastPromise, slowPromise])
  .then(value => {
    console.log(value); // "Fast"
  });

// Async/await (ES2017)
async function fetchData() {
  try {
    console.log("Fetching data...");

// 1. Use const by default, let when needed, avoid var
const PI = 3.14159; // Use const for values that don't change
let counter = 0; // Use let for values that change
// Avoid var due to function scope issues

// 2. Use strict equality (=== and !==) over loose equality
if (value === 'test') { } // Good
if (value == 'test') { } // Avoid

const oldGreeting = 'Hello, ' + name + '!'; // Avoid

// 2. Debounce frequent events
function debounce(func, delay) {
  let timeoutId;
  return function(...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func.apply(this, args), delay);
  };
}

// 3. Throttle expensive operations
function throttle(func, limit) {
  let inThrottle;
  return function(...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

const rateLimit = {};
function isRateLimited(ip) {
  const now = Date.now();
  if (!rateLimit[ip]) {
    rateLimit[ip] = [];
  }
  
  // Remove old entries (last minute)
  rateLimit[ip] = rateLimit[ip].filter(time => now - time < 60000);
  
  if (rateLimit[ip].length >= 10) { // 10 requests per minute
    return true;
  }
  
  rateLimit[ip].push(now);
  return false;
}
